CREATE TABLE Orders (
    OrderID INT,
    CustomerName VARCHAR2(20),
    Product VARCHAR2(20),
    Quantity INT
);

-- Create the Customers table
CREATE TABLE Customers AS
SELECT DISTINCT OrderID, CustomerName
FROM Orders;

-- Add a primary key to the OrderID column in the Customers table
ALTER TABLE Customers
ADD CONSTRAINT pk_OrderID PRIMARY KEY (OrderID);
DESC Customers;

-- Create the OrderDetails table
CREATE TABLE OrderDetails AS
SELECT OrderID, Product, Quantity
FROM Orders;

-- Add a foreign key to the OrderID column in the OrderDetails table
ALTER TABLE OrderDetails
ADD CONSTRAINT fk_OrderID FOREIGN KEY (OrderID)
REFERENCES Customers(OrderID);
DESC OrderDetails;
